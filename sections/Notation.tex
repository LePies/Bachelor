\begin{itemize}
    \item [] \textbf{Symbols}
    \begin{AutoMultiColItemize}
        \item [$t$:] Time
        \item [$m$:] Mass
        \item [$V$:] Volume
        \item [$\rho$:] Densisty
        \item [$v$:] Flow velocity
        \item [$x$:] Position $($ Cartesian $)$
        \item [$\nu$:] Normal vector
        \item [$f$:] Force density $($ Force $\div$ Volume $)$
        \item [$\varrho$:] Momentum
        \item [$F$:] Force
        \begin{itemize}
            \item [$F_m$:] force on budy
            \item [$F_s$:] force on surface
        \end{itemize}
        \item [$P$:] minus fluid stress tensor $($ ??? $)$
        \item [$p$:] Pressure
        \item [$p'$:] Pressure fluctuation $($ ??? $)$
        \item [$a_f$:] acceleration of force $($ eg. gravitation acceleration $)$
        \item [$\tau$:] deviatoric stress tensor $($ ??? $)$
        \item [$\eta$:] dynamic viscosity $($ ??? $)$

        \item [$K_h$:] Heat conduction
        \item [$s$:] Entropy

        \item [$\eye$:] Identity matrix
        \item [$I$:] Intensity $($ Pressure $\div$ Area $)$
        \item [$u'$:] Particle velocity
        \item [$\omega$:] Circula wave frequency
        \item [$\lambda$:] Wave length
    \end{AutoMultiColItemize}

    \item [] \textbf{Operators}
    \begin{AutoMultiColItemize}
        \item [$\nabla \cdot \circ = $] $\text{div}(\circ) = \sum_{i} \frac{\partial}{\partial x_i} \circ$
        \item [$\Dt{\circ}\; =$] $\partt{\circ} + v \nabla\cdot(\circ\cdot v)$
        \item [$\dot{\circ} =$] $\frac{\dev}{\dev V} \Dt{}\circ$
        \item [$\circ_{rms} = $] $\sqrt{\frac{1}{n}||\circ||^2}, \qquad \circ \in \mathbb{R}^n$
        \item [$\Delta \circ = $] $\nabla \cdot (\nabla \cdot \circ)$
    \end{AutoMultiColItemize}
\end{itemize}
